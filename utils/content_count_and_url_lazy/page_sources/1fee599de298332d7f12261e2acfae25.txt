                                                    // To reduce unintended page scroll during a horizontal swipe block the default handler when it looks like the user
                                                    // is in the process of performaing a horizontal scroll

                                                        event.preventDefault();

                                                        event.preventDefault();

                                                    if ( xdiff > 0 )

                                                    }
