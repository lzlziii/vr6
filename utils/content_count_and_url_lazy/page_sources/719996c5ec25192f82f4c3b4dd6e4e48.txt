<script type="text/javascript">

    $(document).ready(function() {

        //Vertical Tab
        $('#parentVerticalTab').easyResponsiveTabs({
            type: 'vertical', //Types: default, vertical, accordion
            width: 'auto', //auto or any width like 600px
            fit: true, // 100% fit in a container
            closed: 'accordion', // Start closed if in accordion view
            tabidentify: 'hor_1', // The tab groups identifier
            activate: function(event) { // Callback function if tab is switched
                var $tab = $(this);
                var $info = $('#nested-tabInfo2');
                var $name = $('span', $info);
                $name.text($tab.text());
                $info.show();
            }
        });
		
		$('#about-newsletter > input[type="submit"]').bind('click', function ( event ) {
			if ( event ) event.preventDefault();
			register($('#about-newsletter'), $('#about-newsletter-feedback'));
		});
		$('#contact-newsletter > input[type="submit"]').bind('click', function ( event ) {
			if ( event ) event.preventDefault();
			register($('#contact-newsletter'), $('#contact-newsletter-feedback'));
		});
    });
	
	function register($form, $feedback) {
		var $email = $form.find('input[type=text]')[0];
		var $button = $form.find('input[type=submit]')[0];
		var $formType = $form.attr('method');
		var $formURL = $form.attr('action');
		var $formData = $form.serialize();
		
		toggleForm($form, $email, $button);
		$feedback.html("");
		$feedback.attr("class", "");
		$.ajax({
			type: $formType,
			url: $formURL,
			data: $formData,
			cache       : false,
			dataType    : 'json',
			contentType: "application/json; charset=utf-8",
			error       : function(err) { 
				toggleForm($form, $email, $button);
				$feedback.html("There was an issue submitting your email :(<br> If the problem persists, please contact us at contact@devioustech.com");
				$feedback.addClass('feedback-error');
			},
			success     : function(data) {
				
				if (data.result != "success") {
					toggleForm($form, $email, $button);
					
					if(data.msg.indexOf("already subscribed") > -1) {
						$feedback.html("The entered email is already subscribed to our newsletter.<br> If you think this might be a technical issue, please contact us at contact@devioustech.com");
					}
					else {
						$feedback.html("The entered email doesn't appear to be valid :(<br> If you think this might be a technical issue, please contact us at contact@devioustech.com");
					}
					
					$feedback.addClass('feedback-error');
				} else {
					$feedback.html("Thank you for subscribing :)<br> Please check your inbox for a confirmation email.");
					$feedback.addClass('feedback-success');
				}
			}
		});
	}
	
	function toggleForm($form, $email, $button) 
	{
		if($email.disabled == false)
		{
			$form.addClass('form-inactive');
			$email.disabled = true;
			$button.disabled = true;
		}
		else
		{
			$form.removeClass('form-inactive');
			$email.disabled = false;
			$button.disabled = false;
		}
	}
	
</script>
