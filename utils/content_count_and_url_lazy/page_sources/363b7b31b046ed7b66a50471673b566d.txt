        function onChangeToElementTree($element, changeFunction) {
            // Watch for changes on an element tree, and call changeFunction when it happens.
            // Requires the browser to support MutationObserver API.

            var MutationObserver = window.MutationObserver || window.WebKitMutationObserver;
            if (!MutationObserver) return;

            var myObserver = new MutationObserver(changeFunction);
            var obsConfig = {
                childList: true,
                characterData: true,
                attributes: true,
                subtree: true
            };

            //--- Add a target node to the observer. Can only add one node at a time.
            $element.each(function() {
                myObserver.observe(this, obsConfig);
            });
