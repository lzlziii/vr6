<script>
    const currentLang = "";

    const getLangId = lang => {
        switch (lang) {
            case 'en':
                return '1';
            case 'ua':
                return '2';
        }
    }

    const select = new Select('#select', {
        placeholder: '',
        selectedId: getLangId(currentLang),
        data: [
            {
                id: '1',
                value: 'EN',
                flag: '../img/flags/en.svg',
            },
            {
                id: '2',
                value: 'UA',
                flag: '../img/flags/ua.svg',
            }
        ],
        onSelect(item) {
            document.location.search = `lang=${(item.value).toLowerCase()}`;
        }
    });
</script>
