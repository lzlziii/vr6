            var pages = new function() {
                var urlhistory = [];
                var currenturlindex = -1;
                var currentcontent = document.querySelector(".content[currentcontent]");
                var othercontent = document.querySelector(".content:not([currentcontent])");
                var iframestage = document.querySelector("#iframe-stage");

                currentcontent.addEventListener("transitionend", function() {
                    iframestage.removeAttribute("transitioning");
                });

                function show(advance) {
                    var url = urlhistory[currenturlindex];
                    var menuentry = document.querySelector('#menubar a[href="' + url + '"]');
                    history.pushState({url: url}, "", url.indexOf("http") < 0 ? url.replace(/\/?index\.(html|php)$/, "") : menuentry ? menuentry.innerText : "external");
                    try {
                        othercontent.firstChild.src = url;
                        var currentIDoc = currentcontent.querySelector("iframe").contentWindow.document;
                        Array.prototype.forEach.call(currentIDoc.querySelectorAll("video, audio"), function(element) {
                            element.pause();
                        });
                        var pC = currentIDoc.querySelector("#playerContainer");
                        pC !== null ? pC.click() : 0;
                    } catch(e) {
                        
                    }
                    iframestage.setAttribute("advance", "" + advance);
                    var refresh = othercontent.offsetHeight;
                    setTimeout(function() {
                        refresh = othercontent.offsetHeight;
                        othercontent.setAttribute("currentcontent", "");
                        currentcontent.removeAttribute("currentcontent");
                        var switchcontent = othercontent;
                        othercontent = currentcontent;
                        currentcontent = switchcontent;
                        iframestage.setAttribute("transitioning", "");
                    }, 0);
                }

                function showprevious() {
                    currenturlindex--;
                    show(false);
                }

                function shownext() {
                    currenturlindex++;
                    show(true);
                }

                function shownew(url) {
                    currenturlindex++;
                    urlhistory = urlhistory.slice(0, currenturlindex);
                    urlhistory.push(url);
                    show(true);
                }

                this.init = function(startpage) {
                    startpage = startpage.substr(1);
                    if(document.querySelector('#menubar a[href="' + startpage + '"]') === null) {
                        startpage = startpage.replace(/\/?index.html/, "");
                        Array.prototype.forEach.call(document.querySelectorAll("#menubar a[href]"), function(element) {
                            if(element.innerText === startpage) {
                                startpage = element.getAttribute("href");
                            }
                        });
                    }
                    currentcontent.firstChild.src = startpage;
                    urlhistory.push(startpage);
                    currenturlindex = 0;
                    
                    history.replaceState({url: startpage}, document.title, location.href);
                    iframestage.setAttribute("advance", "true");

                    this.load = function(url) {
                        if(url === urlhistory[currenturlindex]) {
                            return;
                        }
                        if(currenturlindex > 0 && url === urlhistory[currenturlindex - 1]) {
                            showprevious();
                        } else if(currenturlindex < urlhistory.length - 1 && url === urlhistory[currenturlindex + 1]) {
                            shownext();
                        } else {
                            shownew(url);
                        }
                    };

                    this.back = function() {
                        showprevious();
                    };
                    
                    delete this.init;
                };
            };
            function letLoad(event) {
                var content = this;
                setTimeout(function() {
                    // create hash to detect back navigation within iframe src
                    // use timeout after load for (better) application of hash value(, in particular on back navigation)
                    try {
                        content.contentWindow.location.hash = location.hash.substring(1).length ? location.hash.substring(1) : "!";
                        content.contentWindow.onhashchange = function(event) {
                            if(this.location.hash.length < 2) {
                                pages.back();
                            }
                        };
                    } catch(e) {
                        
                    }
                }, 0);
                
                try {
                    document.title = this.contentWindow.document.title;
    
                    function load(event) {
                        parent.pages.load(this.getAttribute("href"));
                        event.preventDefault();
                    }
                    Array.prototype.forEach.call(this.contentWindow.document.querySelectorAll('a[href]'), function(element) {
                        if((element.href.indexOf("//") === 0 || element.href.indexOf("://") > -1) && element.href.match(/(^\/\/|\:\/\/)uani\.de(\/|$)/) === null) {
                            element.title = "external link";
                        } else {
                            if(element.target !== "_blank") {
                                element.addEventListener("click", load);
                            }
                        }
                    });
                } catch(e) {
                    
                }
            }
            Array.prototype.forEach.call(document.querySelectorAll(".content iframe"), function(element) {
                element.addEventListener("load", letLoad);
            });

                }

            });
