            const buttons = document.querySelectorAll('.btn');
            const containers = document.querySelectorAll('.text-version');

            // Set version-0 as the default active version on page load
            const defaultButton = document.querySelector('.btn.version-0');
            const defaultContainer = document.querySelector('.version-0');
            defaultButton.classList.add('active', `version-${defaultButton.getAttribute('data-version')}`);
            defaultContainer.classList.add('active');

            buttons.forEach(button => {
              button.addEventListener('click', () => {
                const version = button.getAttribute('data-version');
                buttons.forEach(btn => {
                  btn.classList.remove('active', `version-${btn.getAttribute('data-version')}`);
                });
                button.classList.add('active', `version-${version}`);
                containers.forEach(container => {
                  if (container.classList.contains(`version-${version}`)) {
                    container.classList.add('active');
                  } else {
                    container.classList.remove('active');
                  }
                  if (container.classList.contains('version-0') && version !== '0') {
                    container.classList.remove('active');
                  }
                });
                buttons.forEach(btn => {
                    if (btn !== button) {
                      btn.classList.add(`version-${btn.getAttribute('data-version')}`);
                    }
                });
              });
            });
