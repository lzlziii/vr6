		    if ( url.match(new RegExp('/en/')) ) {

		    if ( url.match(new RegExp('/en/')) ){
			    const url_ja = url.replace('/en/', '/');
			    location.href = url_ja;
		    // URLに「/en/」が含まれない場合は日本語サイトのため、
		    //トップページと、そうでない場合で条件分岐してURLを英語サイトのものに変更
		      } else {
		    if ( url == beforeDomain ) {
			    location.href = url + 'en/';
		      } else {
			    url = url.replace(lastDir + '/' , '')
