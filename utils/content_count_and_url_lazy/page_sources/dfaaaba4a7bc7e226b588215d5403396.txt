            idArray.push(jQuery(this).attr("id"));

            jQuery('.' + value).find(".total-slides").html('0' + totalItems);
            console.log(totalItems);
            var currentIndex = jQuery('#' + value + ' .carousel-item.active').index() + 1;
            jQuery('#' + value).bind('slid.bs.carousel', function() {
                var currentindex = jQuery('#' + value).find('.active').index() + 1;
                jQuery('.' + value).find(".count").html('0' + currentindex);
                jQuery('.' + value).find(".crousel-dot").removeClass("active");
                jQuery('.' + value).find('.crousel-dot:nth-child('+currentindex+')').addClass("active");
            });
        });
    }

        jQuery(".letter-container").addClass('trans');

<script>
    if (jQuery(".looking-forward-main").length) {
        var hieghtThreshold = jQuery(".looking-forward-main").offset().top + jQuery(".looking-forward-main").height() - 500;
        var games = jQuery(".going-main").offset().top + jQuery(".going-main").height() - 500;
        jQuery(window).scroll(function() {
            var scroll = jQuery(window).scrollTop();
            if (scroll >= hieghtThreshold) {
                jQuery(".round-ico").addClass('ico-add');
                jQuery(".ico-triangle").addClass('ico-add');
            }
            if (scroll >= games) {
                jQuery(".letter-container").addClass('trans');
            }

        });
        // jQuery(".card-trigger, .member .member-image").click(function() {
        //     jQuery(this).parents(".member").find(".flipcard-holder").toggleClass("is-flipped");
        //     jQuery(this).parents(".member").find(".card-trigger").toggleClass("_active");
        // })

        jQuery('.card-trigger, .member .member-image').hover(function() {
            jQuery(this).parents(".member").find(".flipcard-holder").addClass("is-flipped");
            jQuery(this).parents(".member").find(".card-trigger").addClass("_active");
        },function() {
            jQuery(this).parents(".member").find(".flipcard-holder").removeClass("is-flipped");
            jQuery(this).parents(".member").find(".card-trigger").removeClass("_active");
        });
    }
</script>

<script>
    

    if (jQuery(".testimonial-crousel-wrapper").length > 0) {
        jQuery(".testimonial-slider-main").owlCarousel({
            center: true,
            items: 1,
            loop: true,
            margin: 40,
            stagePadding: 0,
            autoHeight: false,
            dots: true,
            responsive: {
                767: {
                    items: 2,
                },
            },
        });
        var cursor = document.querySelector(".cursor");
        var cursorinner = document.querySelector(".cursor2");
        var a = document.querySelectorAll(".play-btn");
        var section = document.querySelector(".testimonial-crousel-wrapper");

        section.addEventListener("mousemove", function(e) {
            var x = e.clientX;
            var y = e.clientY;
            cursor.style.transform = `translate3d(calc(${e.clientX}px - 50%), calc(${e.clientY}px - 50%), 0)`;
        });

        section.addEventListener("mousemove", function(e) {
            var x = e.clientX;
            var y = e.clientY;
            cursorinner.style.left = x + "px";
            cursorinner.style.top = y + "px";
        });

        section.addEventListener("mousedown", function() {
            cursor.classList.add("click");
            cursorinner.classList.add("cursorinnerhover");
        });

        section.addEventListener("mouseup", function() {
            cursor.classList.remove("click");
            cursorinner.classList.remove("cursorinnerhover");
        });

        section.addEventListener("mouseleave", function() {
            cursor.classList.add("added");
            cursorinner.classList.add("added");
        });
        section.addEventListener("mouseover", function() {
            cursor.classList.remove("added");
            cursorinner.classList.remove("added");
        });
        a.forEach((item) => {
            item.addEventListener("mouseover", () => {
                cursor.classList.add("hover");
            });
            item.addEventListener("mouseleave", () => {
                cursor.classList.remove("hover");
            });
        });
    }
    jQuery(".careers-head .letter-container").addClass("trans");
   
</script>
