function convertToUserTimezoneAndReplace() {
  // Get all HTML elements with the class name "converttousertimezone"
  const elements = document.getElementsByClassName('converttousertimezone');

  // For each element
  for (let i = 0; i < elements.length; i++) {
    // Get the current text content, assuming it's a date string
    const utcDateStr = elements[i].textContent;

    // Convert this date to the user's local timezone
		const date = moment.utc(utcDateStr, 'YYYY-MM-DD hh:mm a').local().toDate();

    // Format the date to a readable string
    const dateString = date.toString();

    // Replace the text content of the element with the converted date
    elements[i].textContent = dateString;
  }
}
