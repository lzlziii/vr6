            $carousel.after(temp).find(".item").each(function(i) {
                var index = (i + 1);
                var itemToClone = $(this);

                    itemToClone = itemToClone.next();
                    // wrap around if at end of item collection

                $('.carousel-showmanymoveone.hidden').find('.fancybox').eq(index - 1).trigger("click");
