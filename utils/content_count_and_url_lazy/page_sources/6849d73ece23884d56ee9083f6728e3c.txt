	// this is the id of the form
function validateEmail($email) {
	var emailReg = /^([\w-\.]+@([\w-]+\.)+[\w-]{2,4})?$/;
	return emailReg.test( $email );
}
function formsubmit(){
	$('#formresponse').empty().show();
	var url = "sendemail.php";	
	var name = $('#name').val();
	var email = $('#email').val();
	var subject = $('#subject').val();
	var mobile = $('#mobile').val();
	var message = $('#message').val();
	if(name == ''){
		$('#formresponse').html('Please enter name..!').css('color','red');
	} else if(email == ''){
		$('#formresponse').html('Please enter email..!').css('color','red');
	} else if(email != '' && !validateEmail(email) ){
		$('#formresponse').html('Please enter valid email..!').css('color','red');
	} else if(mobile == ''){
		$('#formresponse').html('Please enter mobile..!').css('color','red');
	} else if(subject == ''){
		$('#formresponse').html('Please enter subject..!').css('color','red');
	} else if(message == ''){
		$('#formresponse').html('Please enter message..!').css('color','red');
	} else {
		$.ajax({
			type: "POST",
			url: url,
			data: {name:name, email:email, mobile:mobile, subject:subject, message:message},
			success: function(data){			
				$('#formresponse').html(data).css('color','green');
				$('#name, #mobile, #email, #subject, #message').val('');
			}
		});
	}
}
setInterval(function(){ $('#formresponse').empty().hide(); }, 60000);
</script>
