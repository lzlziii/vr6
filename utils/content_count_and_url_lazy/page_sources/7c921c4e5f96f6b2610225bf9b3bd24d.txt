			$(document).ready(function() {
				$('div.question .answers .answer').on('click', function() {
					// Colour the choice depending on if they picked right or wrong
					$(this).filter(".right").css("background-color", "blue");
					$(this).filter(".wrong").css("background-color", "red");
					
					// Remove the colour from the other choices
					$(this).siblings().css("background-color", "white");
					
					
					// Select the feedback for that answer
					answerReveal = $(this).parent().parent().children(".reveals").children(".reveal:eq(" + $(this).index() + ")");
					answerReveal.css("display","block"); // Show the feedback
					answerReveal.siblings(".reveal").css("display","none"); // Hide any other feedback from other answers to this question
					answerReveal.siblings(".quote").css("display","block"); // Show the relevant quote!
				});
			});
