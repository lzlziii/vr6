<script>
    $(function () {
        $('.dropdown-menu').each(function () {
            var $dropdownMenu = $(this);
            var $dropdownMenuItems = $dropdownMenu.find('li');

            if ($dropdownMenuItems.length === 0) {
                $dropdownMenu.parent().removeClass('dropdown');
                $dropdownMenu.parent().find('.dropdown-toggle').removeAttr('data-bs-toggle').removeClass('dropdown-toggle');
                $dropdownMenu.remove();
            } else {
                $dropdownMenu.parent().find('.dropdown-toggle').attr('href', 'javascript:void(0);');
            }
        });

        // make navbar item active based on url without http, domain and query string
        var activePage = window.location.pathname.substring(window.location.pathname.lastIndexOf('/'));

        $('.navbar-nav li a').each(function () {
            var linkPage = $(this).attr('href');

            if (activePage === linkPage || (activePage === "/" && linkPage === "/home")) {
                $(this).addClass('active');

                if ($(this).hasClass('dropdown-item')) {
                    $(this).closest('.nav-item').find('.dropdown-toggle').addClass('active');
                }
            }
        });
    });
</script>

<script>
    $(function () {
        $('.footer-dropdown-menu').each(function () {
            var $dropdownMenu = $(this);
            var $dropdownMenuItems = $dropdownMenu.find('li');

            if ($dropdownMenuItems.length > 0) {
                $dropdownMenu.parent().find('.footer-dropdown-toggle').attr('href', 'javascript:void(0);');
            }
        });

        $('.footer-dropdown-toggle').on('click', function (e) {
            var $self = $(this);

            if ($self.attr('href') === 'javascript:void(0);') {
                e.preventDefault();

                var $dropdownMenu = $self.parent().find('.footer-dropdown-menu');

                if ($dropdownMenu.hasClass('d-none')) {
                    $('.footer-dropdown-menu').addClass('d-none');
                    $dropdownMenu.removeClass('d-none');
                } else {
                    $('.footer-dropdown-menu').addClass('d-none');
                    $dropdownMenu.addClass('d-none');
                }
            }
        });
    });
</script>
