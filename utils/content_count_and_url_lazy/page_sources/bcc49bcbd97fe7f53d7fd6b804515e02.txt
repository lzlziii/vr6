		},
		saveCounts: function() {
			BX.localStorage.set('helpcounts', this.helpcount, 24*60*60*365);
		},
		setCurrentHelpScreen(n) {
			this.slideNum = n;
		},
		chooseHelpMode() {
			switch(this.mode) {
				case 'hall':
					if(this.helpcount[0] <= this.helpcount[1]) {
						this.slideNum = 0;
					} else {
						this.slideNum = 1;
					}
					break;
				case 'obj2d':
					this.slideNum = 2;
					break;
				case 'obj3d':
					this.slideNum = 3;
					break;
			}
		},
		listenCarousel(){
			const myCarousel = document.getElementById('helpCarousel')
			myCarousel.addEventListener('slide.bs.carousel', event => {
			  console.log('carousel events:: ',event.to);
			  this.carouselSlide = event.to;
			})
		},
		/***
		  * { msg_type: helper, action: select/move/shift/zoom/rotate/reset, type: hall/room/art2d/art3d } (type: hall используется только с select);
		  * { msg_type: hover, action: true/false }.
		  */
		messageFromUE: function(msg) {
			try {
				if(msg.hasOwnProperty('msg_type') && msg.msg_type == 'helper') {
					if(msg.hasOwnProperty('action') && msg.action.length > 0) {
						switch(msg.action) {
							case 'select':
								this.helpcount[1]++;
								break;
							case 'move':
								this.helpcount[0]++;
								break;
							case 'rotate':
								this.helpcount[0]++;
								break;
							case 'shift':
								if(this.mode == 'hall') {
									this.helpcount[0]++;
								} else if (this.mode == 'obj2d'){
									this.helpcount[2]++;
								} else {
									this.helpcount[3]++;
								}
								break;
							case 'zoom':
							case 'reset':
							case 'rotate':
								if (this.mode == 'obj2d'){
									this.helpcount[2]++;
								} else {
									this.helpcount[3]++;
								}
								break;
						}
					}
				}
				if(msg.hasOwnProperty('msg_type') && msg.msg_type == 'info') {
					if(msg.hasOwnProperty('hall') && msg.hall.length > 0) {
						this.mode = 'hall';
					}
					if(msg.hasOwnProperty('obj') && msg.obj.length > 0) {
						this.mode = 'obj2d';
						if(msg.hasOwnProperty('type') && msg.type == 'art3d') {
							this.mode = 'obj3d';
						}
					}
					this.chooseHelpMode();
					if(this.visible) {
						this.carousel.to(this.slideNum);
					} else {
						this.tryToShow(); // if requested
					}
				}
			} catch (error) {
				console.warn(error, nav_data);
			}
		},
		requestInfo: function() {
			if(typeof(iframePlayer) !== 'null' && typeof(iframePlayer.sendNavMsg) == 'function') {
				iframePlayer.sendNavMsg({'location':'get'});
			} else {
				console.warn("##>>> HelpPanelApp.requestInfo(). Can't connect to iframePlayer.");
			}
		}
	},
	computed: {
		images: function() {
			return this.helpimg.map(img => {
				let prefix = this.istouch?'touch':'mouse';
				return `${this.base_path}/icons/${this.lang}/${prefix}/${img}`;
			});
		}
