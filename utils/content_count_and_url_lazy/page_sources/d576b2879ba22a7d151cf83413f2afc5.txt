        function vidRescale() {
            var trailer = document.getElementById("trailer"); 

            var w = $(window).width(),
                h = $(window).height();

            if (w / h > 16 / 9) {
                trailer.width = w;
                trailer.height = w / 16 * 9;
                $('.trailer .screen').css({ 'left': '0px' });
            } else {
                trailer.width = h;
                trailer.height = h/ 16 * 9;
                $('.trailer .screen').css({ 'left': -($('.trailer .screen').outerWidth() - w) / 2 });
            }
        }

        $(window).on('load resize', function () {
            vidRescale();
        });

        function muteTrailer() {
            var muteIcon = document.getElementById("mute_icon");
            muteIcon.classList.toggle("fa-volume-off");
            muteIcon.classList.toggle("fa-volume-up");

            //if (menuIcon.classList.contains("fa-volume-on")) {
            //    muteIcon.classList.toggle("fa-volume-on");
            //} else {
            //    muteIcon.classList.addclass("fa-volume-on");
            //}



            if ($("video").prop('muted')) {
                $("video").prop('muted', false);
                $(this).addClass('unmute-video');

            } else {
                $("video").prop('muted', true);
                $(this).removeClass('unmute-video');
            }
            console.log($("video").prop('muted'))
        }
