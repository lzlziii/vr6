function selected(indexIn)
{
	runSlideshow=false;

	document.getElementById('top_picture_panel').style.backgroundImage = "url('"+picArray[picArrayIndex]+"')";

	document.getElementById('dot1').className = "dot";

}


        
function fn60sec()
{
	if(runSlideshow)
	{
			if(picArrayIndex < picArray.length)
			{
				//alert('running');
				// runs every 5 sec and runs on init.
				document.getElementById('top_picture_panel').style.backgroundImage = "url('"+picArray[picArrayIndex]+"')";

				document.getElementById('dot1').className = "dot";

				var temp = 'dot'+tempNumber;
				document.getElementById(temp).className = "dot_selected";

				picArrayIndex++;
			}
			else
			{
				picArrayIndex = 0;
			}
	}
}
