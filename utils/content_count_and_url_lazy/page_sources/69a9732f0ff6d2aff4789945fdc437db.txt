var currentClickedSpell = document.getElementById("spell-clicked");

function theFunction(e) {
    e.target.classList.add("clicked");
    document.getElementById("spell-name").innerHTML = e.target.alt;
    document.getElementById("spell-description").innerHTML = magic[e.target.alt];
    if(currentClickedSpell && (e.target != currentClickedSpell))
    {
        currentClickedSpell.classList.remove("clicked");
    }
    currentClickedSpell = e.target; 
}
