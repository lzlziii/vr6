  var hybridNav = {
    handleBurgerBtn() {
      $('.tm-hamburger').on('click', function() {
        hybridNav._closeNav(0);
        $('.nav-lv-2-wrapper').removeClass('active')
        $('.h-header').toggleClass('nav-open');
        $(this).toggleClass('collapsed');
      });
    },

    handleBackLink() {
      $('.menu-back-link').on("click", function() {
        $('.nav-lv-2-wrapper').removeClass('active')
        hybridNav._closeNav($(this).data("level"));
      });
    },

    handleLvl1() {
      $('.open-nav').on('click', function() {
        var lvl = $(this).data("level");
        var parent = $(this).closest('li');
        // only remove lvl 1
        if ($(window).width() < 768) {
          parent.hasClass('active') ? hybridNav._closeNav(lvl) : (hybridNav._closeNav(lvl), parent.addClass('active'));
        } else {
          parent.hasClass('active') ? (hybridNav._closeNav(lvl), hybridNav._defaultActive()) : (hybridNav._closeNav(lvl), parent.addClass('active'), hybridNav._defaultActive());
        }
      });
    },

    handleLvl2() {
      $('.open-mega').on( "click", function() {
        var active = $(this).data('mega-trigger')
        var lvl_2 = $(this).closest('.mega-menu').find('.nav-lv-2-wrapper');
        lvl_2.removeClass('active');
        $(this).closest('.mega-menu').find('.nav-lv-2-wrapper[data-mega=' + active + ']').addClass('active');
        $(this).closest('ul').find('li').removeClass('active');
        $(this).closest('li').addClass('active');
      });
    },

    handleToggle() {
      if ($(window).width() > 768) {
        hybridNav._closeMega();
      }
    },

    handleLogin() {
      $(".toggle-login-desktop").on("click", function(){
        var txt = $(this).find('span')
        txt.html() === 'Login' ? txt.html('Close') : txt.html('Login')
      })
    },

    tiggerResize() {
      var isDesktopView;
      $(window).on('resize', function() {
          isDesktopView = $(this).width() > 768;
      });
      
      if (isDesktopView) {
        hybridNav._closeMega();
      }
    },

    _closeMega() {
      var $menu = $('.nav-lv-0');
      $(document).mouseup(function(e) {
        // if the target of the click isn't the container...
        // ... nor a descendant of the container
        if (!$menu.is(e.target) && $menu.has(e.target).length === 0) {
          $('.nav-lv-0 > li').removeClass('active');
          hybridNav._defaultActive();
          $(".toggle-login-desktop").find('span').html('Login')
        }
      });
    },

    _closeNav(lvl) {
      $('.nav-lv-' + lvl).find('> li.active').removeClass('active');
    },

    _defaultActive() {
      $('.nav-lv-1 > li').removeClass('active');
      $('.nav-lv-2-wrapper').removeClass('active');
      for(var i=0; i< $('.nav-lv-1').length; i++) {
        $($('.nav-lv-1')[i]).find('.open-mega').first().parent().addClass('active');
        $($('.nav-lv-1')[i]).next('.nav-lv-2-wrapper').addClass('active');
      }
    },

    setMegaMenuHeight() {
      const menuHeight = "calc(100vh - " + $('.h-header').height() + "px)"
      $('.mega-menu').each(function() {
        $(this).css('max-height', menuHeight)
        $(this).find('.nav-lv-1').each(function() {
          $(this).css('max-height', menuHeight)
        })
      })
    },

    init() {
      hybridNav.handleBurgerBtn();
      hybridNav.handleBackLink();
      hybridNav.handleLvl1();
      hybridNav.handleLvl2();
      hybridNav.handleToggle();
      hybridNav.tiggerResize();
      hybridNav.handleLogin();
      hybridNav.setMegaMenuHeight();
    }
  }

  jQuery( document ).ready(function( $ ) {
    hybridNav.init();
  });
