	$(".download").click(function(){
		$(".o-download-wrap").css('display', 'block');
     	$(".g-download-wrap").css('display', 'none');
      	$(".a-download-wrap").css('display', 'none');
	})
	$(".google").click(function(){
      	$(".o-download-wrap").css('display', 'none');
		$(".g-download-wrap").css('display', 'block');
      	$(".a-download-wrap").css('display', 'none');
	})
	$(".apple").click(function(){
      	$(".o-download-wrap").css('display', 'none');
     	$(".g-download-wrap").css('display', 'none');
		$(".a-download-wrap").css('display', 'block');
	})

});
</script>
<script>
$(document).ready(function(){
	$('.cookie').css('position', 'static');
	// cookie consent
	$('.fs-cc-manager3_component').click(function(){

			$('body').addClass('active');

			if($('body').hasClass('active') === true) {
				$('.cookie').css('position', 'fixed');
              
            
			} else {
				$('.cookie').css('position', 'static');

			}

	});
  
  $('.fs-cc-prefs3_close, .fs-cc-prefs3_button, .fs-cc-prefs3_submit, .fs-cc-prefs3_overlay').click(function(){
  
  		$('body').removeClass('active');
  		$('.cookie').css('position', 'static');
  });

  

});
</script>
