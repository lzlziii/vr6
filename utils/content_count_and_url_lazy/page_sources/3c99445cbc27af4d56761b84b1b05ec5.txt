document.addEventListener('DOMContentLoaded', function() {

    /* SCROLLING */
    if (smoothScroll) {
        smoothScroll.init({ offset: 50 });

        if (window.location.hash) {
            var hash = smoothScroll.escapeCharacters(window.location.hash);
            var toggle = document.querySelector('a[href*="' + hash + '"]');
            var options = { speed: 1000 };
            smoothScroll.animateScroll(hash, toggle, options);
        }

        var nextBtn = document.getElementById('nextBlock');

        var blocks = document.querySelectorAll('.block.__scrollable');
        var data = [];

        for (var i = 0; i < blocks.length; i++) {
            if (blocks[i].id)
                data.push({ id: blocks[i].id, offset: blocks[i].offsetTop });
        }


        var prevBlock = null;
        var nextBlock = blocks[0];

        if (nextBtn) {
            nextBtn.addEventListener('click', function(e) {
                e.preventDefault();
                e.stopPropagation();

                if (nextBlock) {
                    smoothScroll.animateScroll('#' + nextBlock.id);
                }
            });
        }

        var footerTrigger = document.body.offsetHeight - window.innerHeight;

        window.addEventListener('scroll', function() {

            var offset = (window.pageYOffset || document.scrollTop) - (document.clientTop || 0);
            if (!offset) offset = 0;
            offset += 400;

            var currentBlock = null;
            for (var i = 0; i < data.length; i++) {
                if (data[i].offset >= offset) {
                    currentBlock = data[i - 1];
                    nextBlock = data[i];
                    break;
                }

                if (i === data.length - 1) {
                    currentBlock = data[i];
                    nextBlock = data[0];
                }
            }

            if (currentBlock !== prevBlock) {
                var links = document.querySelectorAll('.mmenu_list_item_link');
                for (var i = 0; i < links.length; i++) {
                    links[i].classList.remove('__active');
                }
                if (currentBlock) {
                    var link = document.querySelector('a.mmenu_list_item_link[href="#' + currentBlock.id + '"]');
                    link.classList.add('__active');
                }
                prevBlock = currentBlock;
            }
        });

        /* LANG PICKER */
        var langPicker = document.querySelector('.lang-picker');
        var langPickerItem = document.querySelector('.lang-picker_item.__current');
        if (langPickerItem) {
            langPickerItem.addEventListener('click', function(e) {
                e.preventDefault();
                e.stopPropagation();
                if (langPicker) {
                    langPicker.classList.toggle('__hidden');
                }
            });
        }

        /* SCREENSHOTS */
        var screenshots = document.querySelector('.screenshots_list');
        if (screenshots) {
            var bigImage = document.querySelector('.screenshots_big_image');
            screenshots.addEventListener('click', function(e) {
                e.preventDefault();
                if (e.target && e.target.matches('a.screenshots_thumb img')) {
                    e.stopPropagation();
                    var link = e.target.parentElement;
                    var imgPath = link.href;
                    if (imgPath !== bigImage.src) {
                        bigImage.src = imgPath;
                        var current = document.querySelector('.screenshots_thumb.__current');
                        if (current) current.classList.remove('__current');
                        link.classList.toggle('__current');
                    }
                }
            }, false);
        }
    }
});
