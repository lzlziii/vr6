	//This will take place prior to anything else in order to decide whether this is a browser / mobile based platform
    var uagent = navigator.userAgent;

    if((uagent.match(/iPad/i)) || (uagent.match(/iPhone/i)) || (uagent.match(/Android/i))){
        
		console.log("Mobile platform detected");
		window.location.href = 'https://dreamchannel.io';
		
    }else{
        
		console.log("Browser based platform detected");
        
		// Window name is required for window switching / focus scripts to be able to do there thing. eg. dc_social_network or dc_central_world
		// Important change added by RC

		window.name = windowName;
		
		var windowObjectReference = null; // global variable
		
		var unityInstance = UnityLoader.instantiate("unityContainer", "Build/export.json", {onProgress: UnityProgress});
		function UnityProgress(unityInstance, progress) {
			if (!unityInstance.Module) {
				return;
			}
		const loader = document.querySelector("#loader");
		if (!unityInstance.progress) {
			const progress = document.querySelector("#loader .progress");
			progress.style.display = "block";
			unityInstance.progress = progress.querySelector(".full");
			loader.querySelector(".spinner").style.display = "none";
		}
		unityInstance.progress.style.transform = `scaleX(${progress})`;
		if (progress === 1 && !unityInstance.removeTimeout) {
		  unityInstance.removeTimeout = setTimeout(function() {
			  loader.style.display = "none";
		  }, 2000);
		}
    }
  }
