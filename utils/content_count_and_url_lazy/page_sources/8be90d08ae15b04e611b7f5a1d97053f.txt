function elementScrolled(elem)
{
    var docViewTop = $(window).scrollTop();
    var docViewBottom = docViewTop + $(window).height();
    var elemTop = $(elem).offset().top;
    var elemBot = elemTop+$(elem).height();
    return ((elemTop+$(elem).height()/5 <= docViewBottom) && (elemTop+$(elem).height()/5 >= docViewTop)) || ((elemBot-$(elem).height()/5 >= docViewTop) && (elemBot-$(elem).height()/5 <= docViewBottom));
}
