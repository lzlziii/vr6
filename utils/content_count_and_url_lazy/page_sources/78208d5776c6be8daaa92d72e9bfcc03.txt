	$(window).ready(function () {
		let lastclose;
		let limen = 86400000;
		let now;
		let wpobject;

		function fnc_viewtoast() {
			wpobject = new Waypoint({
				element: $('body'),
				handler: function(direction) {
					if(direction === "down"){
						$('.compornent_toast_banner').addClass('active');
					} else {
						$('.compornent_toast_banner').removeClass('active');
					}
				},offset: '-200px'
			});
		}

		function fnc_saveload() {
			now = (new Date()).getTime();
			lastclose = {
				"closetime":now,
				"closeflag":false
			}
			let closedata = JSON.parse(localStorage.getItem('agitbnrclose'));
			if(closedata==null){
				localStorage.setItem('agitbnrclose',JSON.stringify(lastclose));
			} else {
				lastclose = closedata;
			}

			let check= (now > (lastclose.closetime + limen));

			if(!lastclose.closeflag){
				fnc_viewtoast();
			} else if (check){
				fnc_viewtoast();
				localStorage.removeItem('agitbnrclose');
			} else {

			}
		}
		fnc_saveload();


		$('.compornent_toast_close').on('touchstart click', function(event){
			$(this).off();
			$('.compornent_toast_banner').removeClass('active');
			lastclose.closeflag = true;
			lastclose.closetime = now;
			localStorage.setItem('agitbnrclose',JSON.stringify(lastclose));
			wpobject.destroy();
		})

		let charadata;
		$.ajax({
			url: 'https://tokyochronos.com/en/assets/json/compornent_toast.json',
			type: 'GET',
			dataType: "json",
		}).done(function (data){
			charadata= data[0];
			fnc_appendtoast();
		}).fail(function (data){
		});
		
		function fnc_appendtoast() {
			let roll = Math.floor(Math.random() * (Object.keys(charadata).length + 1 - 1)) + 1;
			let count = 0;
			for (let key in charadata) {
				count++;
				if(count==roll){
					$('.compornent_toast_banner a').attr('href', ('https://discord.com/invite/QWJpgNRs4n/?tc_' + charadata[key].no));
					$('.compornent_toast_banner .chara_icon').addClass('vis_'+charadata[key].no);
					$('.compornent_toast_banner dd p').text((charadata[key].セリフ));
					gtag('event', 'click', {'event_category': 'compornent_toast', 'event_label': key});
				}
			}
		}
	});
