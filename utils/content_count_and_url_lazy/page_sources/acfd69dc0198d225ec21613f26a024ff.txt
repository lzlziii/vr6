    <script>function openMenu() {
                var menu = document.getElementById("menu-cover");
                var burger = document.getElementById("burger-menu");
                var links = document.getElementById("menu-links");

                if (menu.classList.contains('show-menu')) {
                    menu.classList.remove('show-menu');
                    burger.classList.remove('rotate-menu');
                    menu.classList.add('hide-menu');
                    links.classList.remove('links-animator');
                    links.classList.add('links-hider');


                }

                else {
                    menu.classList.remove('hide-menu');
                    menu.classList.add('show-menu');
                    burger.classList.add('rotate-menu');
                    links.classList.remove('links-hider');
                    links.classList.add('links-animator');


                }
            }
        </script>
