    const languageSwitcher = document.querySelector("#language-switcher");
    const socialIcons = document.querySelector("#social-icons");

    const animatedElements = [flyoutHeader, flyoutList, languageSwitcher, socialIcons];

    const appsContent = `

        <li><a href="/en/support/">Support</a></li>
        <li class="sub"><a href="/en/support/tutorials-manuals/">Tutorials and Manuals</a></li>
        <li class="sub"><a href="/en/technical-specifications/">Technical Specifications</a></li>
        <li class="sub"><a href="/en/support/faq/">FAQ</a></li>

    `;
    const aboutContent = `
        <li><a href="/en/newsroom/">Newsroom</a></li>
        <li><a href="/en/careers/">Careers</a></li>
        <li><a href="/en/contact/">Contact</a></li>
    `;
    const contentArray = [downloadContent, appsContent, supportContent, aboutContent];

    const languageContent = languageSwitcher.innerHTML;
    const socialIconsContent = socialIcons.innerHTML;

    const handleEnter = () => {
        navbar.style.height = `${flyoutContent.offsetHeight + 38}px`;
        navbar.classList.add("show");
        curtain.classList.add("show");

        //animate text based on if the navbar was open or not
        if (flyoutHeader.classList.contains("initial")) {
            animatedElements.forEach(el => el.classList.remove("initial"));
        } else {
            animatedElements.forEach(el => el.classList.add("trigger-change"));
            setTimeout(() => {
                animatedElements.forEach(el => el.classList.remove("trigger-change"));
            }, 80)
        }
        clearTimeout(mouseOutTimer);
    }
    const handleLeave = () => {
        mouseOutTimer = setTimeout(() => {
            flyoutList.innerHTML = ``;
            flyoutHeader.innerHTML = ``;
            languageSwitcher.innerHTML = ``;
            socialIcons.innerHTML = ``;
            navbar.style.height = "38px";
            animatedElements.forEach(el => el.classList.add("initial"));
            navbar.classList.remove("show");
            curtain.classList.remove("show");
        }, 300);
    }

    flyoutTriggers.forEach((trigger, index) => {
        trigger.addEventListener("mouseenter", () => {
            //dont animate subsequent mouseenters of the same trigger
            if (!(flyoutHeader.innerHTML == trigger.innerHTML)) {
                flyoutHeader.innerHTML = `${trigger.innerHTML}`;
                languageSwitcher.innerHTML = languageContent;
                socialIcons.innerHTML = socialIconsContent;
                flyoutList.innerHTML = contentArray[index];
                handleEnter();
            }
        });
    });

    navbar.addEventListener("mouseleave", () => {
        handleLeave();
    });
</script>

<script>
    let cssStyles = getComputedStyle(document.body);
    const spinnerWrapper = document.querySelector('.carousel-spinner');
    const spinnerItems = document.querySelectorAll('.carousel-spinner > *');
    const dots = document.querySelectorAll('.carousel-controls .dot');
    const playPauseButton = document.getElementById('play-pause');
    const playPauseButtonImg = document.getElementById('button-img');
    const perView = 3;
    const spinnerLength = spinnerItems.length - perView; // We only count the unique slides.
    let totalScroll = 0;
    const transition = cssStyles.getPropertyValue('--transition');
    const delay = 6000;

    spinnerWrapper.style.setProperty('--per-view', perView);
    for(let i = 0; i < perView; i++) {
        spinnerWrapper.insertAdjacentHTML('beforeend', spinnerItems[i].outerHTML);
    }

    // Create a map of dots - slides
    const dotsSlidesPairs = new Map();
    dotsSlidesPairs.set(document.getElementById('dot-one'), document.querySelectorAll('.slide-one'));
    dotsSlidesPairs.set(document.getElementById('dot-two'), document.querySelectorAll('.slide-two'));
    dotsSlidesPairs.set(document.getElementById('dot-three'), document.querySelectorAll('.slide-three'));
    dotsSlidesPairs.set(document.getElementById('dot-four'), document.querySelectorAll('.slide-four'));
    dotsSlidesPairs.set(document.getElementById('dot-five'), document.querySelectorAll('.slide-five'));

    // Current slide counter done with dots
    let currentDotIndex = 0;
    let isPlaying = true;

    // Highlight the first dot and slide initially
    dots[currentDotIndex].style.backgroundColor = '#000';
    let currentSlides = dotsSlidesPairs.get(dots[currentDotIndex]);
    currentSlides.forEach((slide) => {
        let slideBg = slide.firstElementChild;
        slideBg.style.backdropFilter = "brightness(0.8)";
        slideBg.style.webkitBackdropFilter = "brightness(0.8)"
    });

    let autoScroll = setInterval(scrolling, delay);

    function scrolling() {
        totalScroll++;
        const left = parseInt(cssStyles.getPropertyValue('--left-base'));

        // If totalScroll is equal to spinnerLength + 1, reset totalScroll and transition to create the loop effect
        if(totalScroll == spinnerLength + 1) {
            clearInterval(autoScroll);
            spinnerWrapper.style.transition = "0s";
            totalScroll = 1;
            spinnerWrapper.style.left = `${left}px`;
            autoScroll = setInterval(scrolling, delay)
        }

        // Calculate currentDotIndex based on totalScroll
        currentDotIndex = totalScroll % spinnerLength;

        // Reset all dots and highlight the current dot
        dots.forEach(dot => dot.style.backgroundColor = '#7a7a7a');
        dots[currentDotIndex].style.backgroundColor = '#000';

        // Reset all slides and style the current slide
        spinnerItems.forEach((slide) => {
            let slideBg = slide.firstElementChild;
            slideBg.style.backdropFilter = "brightness(0.5)";
            slideBg.style.webkitBackdropFilter = "brightness(0.5)"
        });
        let currentSlides = dotsSlidesPairs.get(dots[currentDotIndex]);
        currentSlides.forEach((slide) => {
            let slideBg = slide.firstElementChild;
            slideBg.style.backdropFilter = "brightness(0.8)";
            slideBg.style.webkitBackdropFilter = "brightness(0.8)"
        });

        // Move the carousel
        const widthEl = document.querySelector('.carousel-spinner > :first-child').offsetWidth + 15;
        spinnerWrapper.style.left = `${(totalScroll * widthEl) * -1 + left}px`;
        spinnerWrapper.style.transition = "2s";
    }
    
    dots.forEach((dot, index) => {
        dot.addEventListener('click', () => {
            clearInterval(autoScroll);
            // Update currentDotIndex based on the clicked dot
            // Update totalScroll
            currentDotIndex = index;
            totalScroll = currentDotIndex;

            const left = parseInt(cssStyles.getPropertyValue('--left-base'));
            const widthEl = document.querySelector('.carousel-spinner > :first-child').offsetWidth + 15;
            spinnerWrapper.style.left = `${(totalScroll * widthEl) * -1 + left}px`;
            spinnerWrapper.style.transition = transition;

            // Reset all dots and highlight the current dot
            dots.forEach(dot => dot.style.backgroundColor = '#7a7a7a');
            dots[currentDotIndex].style.backgroundColor = '#000';
            // Reset slides, style current one
            spinnerItems.forEach((slide) => {
                let slideBg = slide.firstElementChild;
                slideBg.style.backdropFilter = "brightness(0.5)";
                slideBg.style.webkitBackdropFilter = "brightness(0.5)"
            });
            let currentSlides = dotsSlidesPairs.get(dots[currentDotIndex]);
            currentSlides.forEach((slide) => {
                let slideBg = slide.firstElementChild;
                slideBg.style.backdropFilter = "brightness(0.8)";
                slideBg.style.webkitBackdropFilter = "brightness(0.8)"
            });

            // Restart autoScroll only if the carousel is currently playing
            if(isPlaying) autoScroll = setInterval(scrolling, delay);

            isPlaying = false;
            clearInterval(autoScroll);
            playPauseButtonImg.src = '/images/icons/play-fill.svg';
        });
    });

    playPauseButton.addEventListener('click', () => {
        if (isPlaying) {
            clearInterval(autoScroll);
            playPauseButtonImg.src = '/images/icons/play-fill.svg';
        } else {
            autoScroll = setInterval(scrolling, delay);
            playPauseButtonImg.src = '/images/icons/pause-fill.svg';
        }
        isPlaying = !isPlaying;
    });
</script>
