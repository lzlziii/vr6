                                                // swipe gesture handling for the media carousel

                                                // TODO: We may want to move all of this to the player class so it has built in touch gesture support.
                                                // First testing here as part of new mobile ux.

                                                var k_nElementWidth = 120;
                                                var k_playerItemClassName = '.highlight_player_item';
                                                var g_nStripWidth = 2760;
                                                var g_nMediaItems = player.m_elemPlayerArea.find( '.highlight_player_item' ).length;
                                                var g_TouchEventStart = null;

                                                // Handle left/right swipe on the media playback by moving forward or backward when the swipe completes
                                                $J('#highlight_player_area').on("touchstart", function (event) {

                                                });
                                                $J('#highlight_player_area').on("touchmove", function (event) {

                                                    if ( typeof g_TouchEventStart == undefined )
                                                        return;

                                                    var xdiff = Math.abs( g_TouchEventStart.originalEvent.changedTouches[0].screenX - event.originalEvent.changedTouches[0].screenX );
                                                    var ydiff = Math.abs( g_TouchEventStart.originalEvent.changedTouches[0].screenY - event.originalEvent.changedTouches[0].screenY );

                                                    // To reduce unintended page scroll during a horizontal swipe block the default handler when it looks like the user
                                                    // is in the process of performaing a horizontal scroll
                                                    if ( xdiff > ydiff && ( typeof event.cancelable !== 'boolean' || event.cancelable ) )
                                                        event.preventDefault();

                                                });
                                                $J('#highlight_player_area').on("touchend", function (eventEnd) {

                                                    if ( typeof g_TouchEventStart.originalEvent.changedTouches[0] == undefined ||
                                                        typeof eventEnd.originalEvent.changedTouches[0] == undefined )
                                                        return;

                                                    var xdiff = g_TouchEventStart.originalEvent.changedTouches[0].screenX - eventEnd.originalEvent.changedTouches[0].screenX;
                                                    var ydiff = g_TouchEventStart.originalEvent.changedTouches[0].screenY - eventEnd.originalEvent.changedTouches[0].screenY;

                                                    // clear out the touch start event
                                                    g_TouchEventStart = null;

                                                    // ignore the swipe if it was short and could've been intended as a tap, or the
                                                    // direction was more vertical than horizontal
                                                    if ( Math.abs( xdiff ) < 20 || Math.abs( ydiff ) > Math.abs( xdiff ) )
                                                        return;

                                                    // we're handling this event
                                                    if ( typeof event.cancelable !== 'boolean' || event.cancelable )
                                                        event.preventDefault();

                                                    // move forward or backward but don't wrap as that would be a jarring user experience
                                                    if ( xdiff > 0 )
                                                    {
                                                        var $NextItem = player.m_activeItem.next( k_playerItemClassName );
                                                        if ( $NextItem.length )
                                                            player.Transition( true );
                                                    }
                                                    else
                                                    {
                                                        var $NextItem = player.m_activeItem.prev( k_playerItemClassName );
                                                        if ( $NextItem.length )
                                                            player.TransitionBack( true );
                                                    }
                                                });

                                                // when the thumbnail strip is swiped move the strip at a granular level
                                                // so it feels active.  The active image does not update because when testing the code to
                                                // make it change the active element it seemed annoying

                                                var g_nMaxScroll = g_nMediaItems > 3 ? ( ( g_nMediaItems - 3 ) * k_nElementWidth ) : 0;

                                                    g_TouchEventStart = event;
                                                    g_nSwipeStartingPosition = parseInt( player.m_elemStripScroll.css( 'left' ) );

                                                    player.StopCycle();
                                                });

                                                    // is in the process of performaing a horizontal scroll

                                                    var ydiff = parseInt( event.originalEvent.changedTouches[0].screenY - g_TouchEventStart.originalEvent.changedTouches[0].screenY );
                                                    if ( Math.abs(ydiff) > Math.abs(xdiff) )
                                                        return;

                                                    // move the scroll strip without changing the selected item
                                                    var nNewScrollPosition = parseInt( g_nSwipeStartingPosition + xdiff );

                                                    // check that we don't scroll past the ends
                                                    if ( nNewScrollPosition > 0 )
                                                        nNewScrollPosition = 0;
                                                    else if ( nNewScrollPosition < (-1 * g_nMaxScroll) )
                                                        nNewScrollPosition = (-1 * g_nMaxScroll);

                                                    // set the new scroll position
                                                    player.m_elemStripScroll.css( 'left', nNewScrollPosition + 'px' );

                                                        event.preventDefault();
                                                });

					非常好的一款漂移模拟游戏。画面不是特别好，但是随着几次更新也有很大进步。不过评价这类游戏主要还是看对方向盘外设的支持程度，包括力反馈的可设置调节选项的丰富程度，细致程度，以及实用程度等的高低和赛车在游戏中漂移过程中与现实汽车漂移大赛中的赛车在漂移过程中行驶的动态轨迹，比如滑行的速度和当时角度的大小，车身的倾斜姿态，上下坡，或是加速减速和手刹对行车姿态的影响的逼真程度的接近与否来予以评价。再比较下同类赛车模拟游戏，比如神力科莎，如果想达到专业改装漂移赛车的大角度滑行，又想用中端的T300或是入门的G29，27这类比较低的回盘速度的盘子，却又想达到非常逼真的游戏效果，很难，或是几乎不可能，除非有非常适合这类盘子的漂移车MOD，以及提前回盘的预判，但是这种游戏体验真的很差，就是和自己过不去。但是该游戏，T300就足够了，如果有更好的盘子比如TSPC,TGT当然更好。因为我发现这个游戏，不像其他模拟游戏，有太多太杂的力反馈的干扰项，比如路面的颠簸的影响，前轮转向不足的影响，车身重量转移的影响，还有最直接的方向盘的阻力影响，就是轻重，中心弹力的影响。因为太多的干扰项会使中低端的方向盘应接不暇，除非是高端盘子，不然选项越多，盘子的实际使用体验越差。不是说神力科莎不好，只是它适合较高端方向盘，那样才能完整体验它的乐趣。简而言之，这游戏可以用比较入门的方向盘体会漂移的乐趣。					<div class="gradient"></div>

				<div class="content">
					<div class="bb_h1">前言</div>这是一款拥有一定游玩门槛的游戏，他并不是《飙酷车神》那种大地图奔驰的爽作，虽然同样以真实赛事作为卖点，但又没有f1那种代入感较高的生涯扮演模式。甚至对普通玩家来说都缺乏像样单机内容。但就如同标题所表达的，这就是一款专注于“漂移体验”的针对向游戏。真实的物理漂移，细致的重量等要素对漂移效果的直接反馈都决定了这是爱好者的宝藏游戏，但对于普通玩家而言，缺乏单机内容和较为平庸的画面都让作品显得不那么诱人。而在操作上想获得最好体验还要依靠专门的方向盘又增加了额外的门槛，总的来说漂移爱好者可以购买，但只是想玩一款爽快赛车游戏的玩家有比本作更好的选择。<br><br><div class="bb_h1">优点</div>+出色的漂移体验<br>虽然初上手可能会觉得本作的车辆有些“飘”。但习惯后本作的漂移无论是操作难度还是实际体验都比较出色，视觉上的烟雾和听觉上的轮胎摩擦声都比较到位。且游戏还加入了较为专业的重量等要素对漂移手感的影响，即便感觉手感不舒服也可以通过较为强大的自定义车辆来调整到自己感觉舒服的程度。<br><br>+清晰明了的车辆改造<br>不同于大部分赛车游戏改造项目上大堆看不懂的数值和品牌，本作的改装十分简明易懂，hp决定了车辆的受创上限。重量则决定了漂移时车辆的偏移程度和拐弯的手感。同时改装系统本身也只需要你攒钱就可以完成而非一般游戏中的各种完成xx赛事获得多少粉丝解锁，对于只想马上投入到纯粹技巧训练中的玩家相当友好。<br><br>+完善的线上内容<br>这基本上是游戏90%的内容所在，玩家可以自由的创建比赛，决定是面向所有玩家还是自己的好友，同时还可以方便的加入其他人的比赛。游戏还内置了一套排名系统，甚至还有直接依据定制规则来的youtube线上直播大赛。且游戏还加入了类似于网游的发言大厅，给人的感觉是制作组想要把游戏打造为一个虚拟的漂移爱好者交流中心。个人的感觉是这种内容比较核心化的游戏确实很容易找到能够长期交流的同好.<br><a class="bb_link" href="https://steamcommunity.com/sharedfiles/filedetails/?id=1978049783" target="_blank" rel="" id="dynamiclink_0">https://steamcommunity.com/sharedfiles/filedetails/?id=1978049783</a><br>（全球玩家的大赛看起来确实让人产生一种“想要看看自己有多少分量”得冲动）<br><br><br><div class="bb_h1">缺点</div>-杂乱的ui界面<br><a class="bb_link" href="https://steamcommunity.com/sharedfiles/filedetails/?id=1978049706" target="_blank" rel="" id="dynamiclink_1">https://steamcommunity.com/sharedfiles/filedetails/?id=1978049706</a><br>（说实话我估计大部分玩家看到本作的这个主菜单ui的第一反应都是太乱了..看起来跟超市的优惠广告一样..）<br><br>-手柄操作存在问题<br>个人使用xboxone精英手柄，在主界面可以正常使用，但进入实际赛道后则无法做出任何反馈，查询讨论区有好几个人遇到了和我一样的问题，但官方都没有给出相对应的解答。<br><br>-基本没有什么单机内容<br>本作几乎全部的游玩内容都是网络相关，游戏在这方面的支持也确实非常到位，无论是世界赛还是邀请好友的功能都做的比较完备。但对于纯单机玩家来说似乎就只能在没有对手的赛道上练习，没有类似生涯或是离线竞赛之类的内容让单机玩家几乎找不到什么能玩的东西。<br><br>-画质不具备竞争力<br>这是我在最高画质下的截图<br><a class="bb_link" href="https://steamcommunity.com/sharedfiles/filedetails/?id=1978049893" target="_blank" rel="" id="dynamiclink_2">https://steamcommunity.com/sharedfiles/filedetails/?id=1978049893</a><br>可以看出本作虽然车辆细节尚可，但赛道环境基本处于上个时代的水平，在如今越发追求画面的竞速类游戏下显得不具备太多的竞争力和吸引力。对比商店截图来说已经有点包装图片仅供参考的味道了..<br><br><div class="bb_h1">一句话总结</div>本作并非以大地图，氮气加速等“爽作”内容为卖点，对于漂移爱好者来说本作提供了拟真与娱乐性平衡的较好的游戏模式，上手也比较简单，但对于视觉系或者是追求爽快的玩家个人十分不推荐入手。<br><br><blockquote class="bb_blockquote">欢迎关注我们的鉴赏家:<a class="bb_link" rel="">走心评测</a>←点击加入，一起探索游戏的乐趣</blockquote>					<div class="gradient"></div>
